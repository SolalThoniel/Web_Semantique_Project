//Filtrer par nom ("Slayers") ici
Select ?c where{
    ?c  rdf:type dbo:Comic.
    ?c rdfs:label ?label.
     Filter(contains(?label, "Slayers"))
}

//Filtrer par auteur ("Takahashi") ici
Select ?c where{
    ?c  rdf:type dbo:Comic.
    ?c dbo:author ?author.
    ?author rdfs:label ?authorName
     Filter(contains(?authorName, "Takahashi"))
}


//Filtrer par type ("Manga") ici
Select ?c where{
    ?c  rdf:type dbo:Comic.
    ?c dbo:type ?type.
    ?type rdfs:label ?typeName
     Filter(contains(?typeName, "Manga"))
}

//Filtrer par contenu du résumé ("Japanese manga series written") ici
Select ?c where{
    ?c  rdf:type dbo:Comic.
    ?c dbo:abstract ?contenu. 
     Filter(contains(?contenu, "Japanese manga series written"))
}

//Combinaison du titre et du contenu
Select ?c where{
    ?c rdf:type dbo:Comic.
    ?c dbo:abstract ?contenu. 
    ?c rdfs:label ?label.
    Filter(contains(?contenu, "Japanese manga series written") && contains(?label, "Ikkoku"))
}


Select distinct ?c ?name where{
    ?c  rdf:type dbo:Manga, dbo:Comic.
    ?c dbp:name ?name.
    ?c dbp:jaRomaji ?romaji.
    ?c dbp:jaKanji ?kanji.
 MINUS {?c dct:subject dbc:Visual_novels.}
 MINUS {?c rdf:type dbo:Film.}
 MINUS {?c dbo:type dbr:Light_novel.}
 filter (contains(lcase(str(?name)),lcase("motsCles")) || contains(lcase(str(?romaji)),lcase("motsCles")) || contains(str(?kanji),lcase("motsCles")) )
}

//recherche par nom de manga
Select distinct ?c ?name where{
    ?c dbo:type dbr:Manga.
    ?c dbp:name ?name.
    ?c dbp:jaRomaji ?romaji.
    ?c dbp:jaKanji ?kanji.
 filter (contains(lcase(str(?name)),lcase("motsCles")) || contains(lcase(str(?romaji)),lcase("motsCles")) || contains(str(?kanji),lcase("motsCles")) )
}

//recherche par nom auteur ou illustrateur
Select distinct ?c where{
    ?c dbo:type dbr:Manga.
    ?c dbp:author ?auteur.
    ?c dbp:illustrator ?illu.
 filter (contains(lcase(str(?auteur)),lcase("motsCles")) || contains(lcase(str(?illu)),lcase("motsCles")))
}
